<script>
  // ── Config ──────────────────────────────────────────
  const APP_ID  = 1089;           // public demo app id
  const SYMBOL  = 'R_100';
  const WS_URL  = `wss://ws.binaryws.com/websockets/v3?app_id=${APP_ID}`;
  const MAX_VISIBLE = 100;        // default ticks shown
 
  // ── State ───────────────────────────────────────────
  let allTicks  = [];
  let maxVisible = MAX_VISIBLE;
  let openPrice  = null;
  let high = -Infinity, low = Infinity;
  let tickCount  = 0;
  let ws;
 
  // ── Clock ────────────────────────────────────────────
  setInterval(() => {
    document.getElementById('utc-time').textContent =
      new Date().toUTCString().slice(17, 25) + ' UTC';
  }, 1000);
 
  // ── Chart setup ──────────────────────────────────────
  const ctx = document.getElementById('chart').getContext('2d');
  const gradient = ctx.createLinearGradient(0, 0, 0, 400);
  gradient.addColorStop(0,   'rgba(0,229,160,0.18)');
  gradient.addColorStop(1,   'rgba(0,229,160,0)');
 
  const chart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: [],
      datasets: [{
        label: 'R_100',
        data: [],
        borderColor: '#00e5a0',
        borderWidth: 1.5,
        backgroundColor: gradient,
        pointRadius: 0,
        tension: 0.1,
        fill: true,
      }]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      animation: false,
      interaction: { intersect: false, mode: 'index' },
      plugins: {
        legend: { display: false },
        tooltip: {
          backgroundColor: '#10141c',
          borderColor: '#1e2430',
          borderWidth: 1,
          titleColor: '#8892a4',
          bodyColor: '#e2e8f0',
          titleFont: { family: 'JetBrains Mono', size: 10 },
          bodyFont:  { family: 'JetBrains Mono', size: 12 },
          callbacks: {
            title: items => items[0].label,
            label: item  => ' ' + item.raw.toFixed(3),
          }
        }
      },
      scales: {
        x: {
          ticks: { color: '#4a5568', font: { family: 'JetBrains Mono', size: 9 }, maxTicksLimit: 8 },
          grid:  { color: '#1e2430' },
        },
        y: {
          position: 'right',
          ticks: { color: '#8892a4', font: { family: 'JetBrains Mono', size: 10 }, maxTicksLimit: 6 },
          grid:  { color: '#1e2430' },
        }
      }
    }
  });
 
  // ── Helpers ──────────────────────────────────────────
  function fmtTime(epoch) {
    const d = new Date(epoch * 1000);
    return d.toUTCString().slice(17, 25);
  }
 
  function updateChart() {
    const slice  = allTicks.slice(-maxVisible);
    chart.data.labels   = slice.map(t => fmtTime(t.epoch));
    chart.data.datasets[0].data = slice.map(t => t.quote);
    chart.update('none');
  }
 
  function setMax(n) {
    maxVisible = n;
    ['100','200','500'].forEach(x => {
      document.getElementById('btn-'+x).classList.toggle('active', x == n);
    });
    updateChart();
  }
 
  // ── WebSocket ─────────────────────────────────────────
  function connect() {
    ws = new WebSocket(WS_URL);
 
    ws.onopen = () => {
      document.getElementById('conn-status').textContent = 'Connected';
      document.getElementById('conn-status').className = '';
 
      // Get last 500 ticks first
      ws.send(JSON.stringify({
        ticks_history: SYMBOL,
        adjust_start_time: 1,
        count: 500,
        end: 'latest',
        start: 1,
        style: 'ticks'
      }));
    };
 
    ws.onmessage = (e) => {
      const msg = JSON.parse(e.data);
 
      // History response
      if (msg.history) {
        const prices = msg.history.prices;
        const times  = msg.history.times;
        allTicks = times.map((t, i) => ({ epoch: t, quote: prices[i] }));
        openPrice = allTicks[0]?.quote;
        high = Math.max(...prices);
        low  = Math.min(...prices);
        tickCount = allTicks.length;
        updateStats(allTicks[allTicks.length - 1]?.quote);
        updateChart();
 
        // Now subscribe to live ticks
        ws.send(JSON.stringify({ ticks: SYMBOL, subscribe: 1 }));
      }
 
      // Live tick
      if (msg.tick) {
        const { quote, epoch } = msg.tick;
        const prev = allTicks[allTicks.length - 1]?.quote;
 
        allTicks.push({ epoch, quote });
        if (allTicks.length > 2000) allTicks.shift();
 
        tickCount++;
        if (quote > high) high = quote;
        if (quote < low)  low  = quote;
 
        updateStats(quote, prev);
        updateChart();
 
        document.getElementById('last-tick').textContent =
          'Last tick: ' + fmtTime(epoch);
      }
 
      if (msg.error) {
        document.getElementById('conn-status').textContent = msg.error.message;
        document.getElementById('conn-status').className = 'error';
      }
    };
 
    ws.onerror = () => {
      document.getElementById('conn-status').textContent = 'Connection error — retrying…';
      document.getElementById('conn-status').className = 'error';
    };
 
    ws.onclose = () => {
      document.getElementById('conn-status').textContent = 'Disconnected — reconnecting…';
      document.getElementById('conn-status').className = 'error';
      setTimeout(connect, 3000);
    };
  }
 
  function updateStats(price, prev) {
    if (!price) return;
    const el = document.getElementById('price');
    el.textContent = price.toFixed(3);
 
    // Flash colour
    if (prev !== undefined) {
      el.classList.remove('flash-up','flash-down');
      void el.offsetWidth;
      el.classList.add(price >= prev ? 'flash-up' : 'flash-down');
    }
 
    // Change vs open
    if (openPrice) {
      const diff = price - openPrice;
      const pct  = ((diff / openPrice) * 100).toFixed(2);
      const sign = diff >= 0 ? '+' : '';
      const chEl = document.getElementById('price-change');
      chEl.textContent = `${sign}${diff.toFixed(3)} (${sign}${pct}%)`;
      chEl.className = 'price-change ' + (diff >= 0 ? 'up' : 'down');
    }
 
    document.getElementById('stat-high').textContent  = high.toFixed(3);
    document.getElementById('stat-low').textContent   = low.toFixed(3);
    document.getElementById('stat-open').textContent  = openPrice?.toFixed(3) ?? '--';
    document.getElementById('stat-ticks').textContent = tickCount;
  }
 
  connect();
</script>
